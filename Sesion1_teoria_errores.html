function delta=error_rel(valor_exacto,valor_aprox)
% Doc de la funci√≥n (ejercicio)
 
% el s√≠mbolo ; es para que no imprima en consola
delta = abs(valor_exacto-valor_aprox) / abs(valor_exacto);
end
 
function S=expo(x,n)
% Doc
 
% S=e^x
S=0;
for k=0:n
% t√©rmino k-√©simo
t_k = x^k/factorial(k);
% agregando en S
S = S + t_k;
end
end
 
function S=fcos(x,n)
% S=FCOS(x,n) devuelve el coseno de x basado en la serie
% cos(x) = 1-x^2/2+x^4/4!-x^6/6!+x^8/8!+...
% Aqu√≠ n+1 indica el n√∫mero de t√©rminos empleados.
%
% Ejemplo:
% >> fcos(2,4)
 
% S=cos(x)
S=0;
for k=0:n
% t√©rmino k-√©simo
t_k = (-1)^k * x^(2*k)/factorial(2*k);
% agregando en S
S = S + t_k;
end
 
end

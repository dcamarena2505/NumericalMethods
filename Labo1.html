Funciones definidas en este live script
function S=expo(x,n)
% Doc
% S=e^x
S=0;
for k=0:n
% término k-ésimo
t_k = x^k/factorial(k);
% agregando en S
S = S + t_k;
end
end
function S=expon(x,n)
% Doc
% S=e^(-x)
S=0;
for k=0:n
% término k-ésimo
t_k = (-1)^k * x^k/factorial(k);
% agregando en S
S = S + t_k;
end
end
function S=fcos(x,n)
% S=FCOS(x,n) devuelve el coseno de x basado en la serie
% cos(x) = 1-x^2/2+x^4/4!-x^6/6!+x^8/8!+...
% Aquí n+1 indica el número de términos empleados.
%
% Ejemplo:
% >> fcos(2,4)
% S=cos(x)
S=0;
for k=0:n
% término k-ésimo
t_k = (-1)^k * x^(2*k)/factorial(2*k);
% agregando en S
S = S + t_k;
end
end

Funciones de usuario
function y=redond(x,n)
% y = REDOND(x,n) devuelve x redondeado a n cifras significativas, aquí
% x es un número real y n es un entero positivo.
 
% Advertencia: mayor error cuando x es complejo, por ejemplo 1+3.98i
 
if x==0
y=0;
else
s=sign(x);
expo=ceil(log(abs(x))/log(10));
mant=exp(log(abs(x))-expo*log(10));
nm=round(mant*10^n)/(10^n);
y=s*nm*10^expo;
end
end
 
function y=suma(a,b,n)
y=redond(redond(a,n)+redond(b,n),n);
end
 
function y=resta(a,b,n)
y=redond(redond(a,n)-redond(b,n),n);
end
 
function y=mult(a,b,n)
y=redond(redond(a,n)*redond(b,n),n);
end
 
function y=div(a,b,n)
y=redond(redond(a,n)/redond(b,n),n);
end
 
function y=raiz(a,n)
y=redond(sqrt(redond(a,n)),n);
end
 
function [x1,x2]=cuadratica(a,b,c,n)
% [x1,x2]=CUADRATICA(a,b,c,n) resuelve la ecuacion a*x^2+b*x+c=0
% considerando n cifras significativas
 
% Discriminante
Delta = resta(mult(b,b,n), mult(4,mult(a,c,n),n), n);
 
x1=div(suma(-b,raiz(Delta,n),n), mult(2,a,n),n);
x2=div(resta(-b,raiz(Delta,n),n), mult(2,a,n),n);
end
